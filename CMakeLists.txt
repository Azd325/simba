PROJECT(simba)
FIND_PACKAGE(Qt4 REQUIRED)

SET(simba_SOURES src/main.cpp src/mainwindow.cpp)
SET(simba_HEADERS src/mainwindow.h)
SET(simba_RESOURCES src/data/data.qrc)
SET(QT_USE_QTWEBKIT TRUE)

QT4_WRAP_CPP(simba_HEADERS_MOC ${simba_HEADERS})
QT4_ADD_RESOURCES(simba_RESOURCES_RCC ${simba_RESOURCES})

INCLUDE(${QT_USE_FILE})
ADD_DEFINITIONS(${QT_DEFINITIONS})

ADD_EXECUTABLE(simba ${simba_SOURCES} 
    ${simba_HEADERS_MOC}
    ${SIMBA_RESOURCES_RCC})
    TARGET_LINK_LIBRARIES(simba ${QT_LIBRARIES})

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})
